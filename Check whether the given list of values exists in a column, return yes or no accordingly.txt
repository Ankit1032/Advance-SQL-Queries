--Check whether the given list of values exists in a column, return yes or no accordingly
SELECT
    e1.employee_id,
    e1.job_id,
    column_value id_job,
    CASE
        WHEN e1.job_id IS NULL THEN
            'NO'
        ELSE
            'YES'
    END          AS does_it_exists
FROM
    sys.odcivarchar2list('SA_REP', 'ST_MAN', 'NotValid', 'FI_ACCOUNT') l1
    LEFT JOIN employees
